```typescript
/** A single routing rule with criteria and target model */
export interface ModelRoutingRule {
  when?: {
    input_tokens_estimate?: string;          // comparison: "<500", ">4000", "<=1000"
    system_prompt_contains?: string[];       // ANY keyword match in system prompt -> rule matches
    no_system_prompt_contains?: string[];    // NONE of these keywords may appear in system prompt
    user_prompt_contains?: string[];         // ANY keyword match in user message content
    no_user_prompt_contains?: string[];      // NONE of these keywords may appear in user message
    agent?: string;                          // literal agent ID or "*" for wildcard
    time_of_day?: string;                    // "HH:MM-HH:MM" range (UTC unless timezone in policy)
    tool_calls_present?: boolean;            // true if request includes tool/function definitions
    conversation_turns?: string;             // comparison: "<3", ">10", "<=5"
    provider?: string;                       // literal provider name match
  };
  route_to: string;                          // target model name or alias (resolved via model_aliases)
  default?: 'passthrough';                   // if set to 'passthrough', this is the fallback rule
}

/** Model route policy — smart model routing per ADR-021 */
export interface ModelRoutePolicy extends PolicyBase {
  type: 'model_route';
  rules: ModelRoutingRule[];
  model_aliases?: Record<string, string>;    // e.g., { cheap: "claude-haiku-4-5-20251001", standard: "claude-sonnet-4-5-20250929" }
  max_downgrade_level?: string;              // alias tier name: routing cannot go below this tier
  routing_opt_out_agents?: string[];         // agents that skip routing entirely (passthrough)
}
```

```typescript
function evaluateModelRoute(
  policy: ModelRoutePolicy,
  context: PolicyRequestContext,
  now?: number,
): ModelRouteResult {
```

**Implementation details:**

1. **Per-agent opt-out check**: If `context.agentId` is in `policy.routing_opt_out_agents`, return passthrough immediately (allowed: true, no routeTo).

2. **Iterate rules in order** (first match wins):
   - For each rule, if `rule.default === 'passthrough'`, return passthrough.
   - If `rule.when` is defined, check ALL criteria (AND logic):
     - `input_tokens_estimate`: Parse the comparison string (e.g., `"<500"`) with a helper `parseComparison(value: string): { op: string, num: number }`. Compare against `context.inputTokensEstimate ?? 0`. Operators: `<`, `>`, `<=`, `>=`, `=`.
     - `system_prompt_contains`: Check if ANY keyword appears (case-insensitive) in `context.systemPrompt`. If systemPrompt is undefined/empty, no match.
     - `no_system_prompt_contains`: Check that NONE of the keywords appear (case-insensitive) in `context.systemPrompt`. If systemPrompt is undefined/empty, this criterion passes (nothing contains the forbidden words).
     - `user_prompt_contains`: Same logic as system_prompt_contains but on `context.userPrompt`.
     - `no_user_prompt_contains`: Same logic as no_system_prompt_contains but on `context.userPrompt`.
     - `agent`: Match literal or `"*"` wildcard against `context.agentId`.
     - `time_of_day`: Parse `"HH:MM-HH:MM"` range. Convert `now` (or Date.now()) to HH:MM in UTC. Check if current time falls within range (support overnight ranges like "22:00-06:00" via OR logic, same as time_window evaluator).
     - `tool_calls_present`: Compare boolean against `context.toolCallsPresent ?? false`.
     - `conversation_turns`: Parse comparison string, compare against `context.conversationTurns ?? 0`.
     - `provider`: Literal match against `context.provider`.
   - If ALL criteria match, this rule fires.

3. **Resolve route_to through model_aliases**: If `rule.route_to` exists as a key in `policy.model_aliases`, resolve to the alias value. Otherwise use `route_to` as a literal model name.

4. **max_downgrade_level enforcement**: If `policy.max_downgrade_level` is set and `policy.model_aliases` is defined:
   - Build a tier order from model_aliases keys (first key = lowest tier, last key = highest tier).
   - Find the tier index of the resolved model and the max_downgrade tier.
   - If the resolved model tier < max_downgrade tier, skip this rule (passthrough instead).
   - If the resolved model is not in aliases (literal model name), allow it (no tier restriction on explicit models).

5. **No match**: If no rules fire, return passthrough (allowed: true, no routeTo).

6. **Return structure**: `ModelRouteResult` with:
   - `allowed: true` always (model_route never denies — it reroutes or passes through)
   - `routeTo`: resolved model string (or undefined for passthrough)
   - `requestedModel`: `context.model`
   - `matchedRuleIndex`: index of the matched rule (or undefined)
   - `policyName`, `policyType: 'model_route'`

**Wire into evaluatePolicy dispatch**: Replace the Phase 6 stub `case 'model_route'` in `evaluatePolicy()` with a call to `evaluateModelRoute(policy, context, now)`.

**Helper function `parseComparison()`**: Extract operator and number from strings like `"<500"`, `">4000"`, `"<=1000"`.
