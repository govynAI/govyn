**1. Create `dashboard/src/types/api.ts`** — TypeScript types matching the proxy API response shapes:

```typescript
// Mirror the proxy's response types for the dashboard
export type TimePeriod = 'hour' | 'day' | 'week' | 'month' | 'all';

// Maps to user-facing period switcher labels
export type DashboardPeriod = 'today' | '7d' | '30d' | 'all';

// Map dashboard period to API period param
export function toApiPeriod(period: DashboardPeriod): string {
  switch (period) {
    case 'today': return 'today';
    case '7d': return 'week';   // Note: proxy doesn't have 'week' in cost-api, but aggregator supports it
    case '30d': return 'month';
    case 'all': return 'all';
  }
}

export interface ModelBreakdown {
  cost: number;
  requests: number;
  inputTokens: number;
  outputTokens: number;
}

export interface AgentCostSummary {
  agentId: string;
  totalCost: number;
  inputCost: number;
  outputCost: number;
  totalInputTokens: number;
  totalOutputTokens: number;
  requestCount: number;
  models: Record<string, ModelBreakdown>;
}

export interface CostsApiResponse {
  period: string;
  generated_at: string;
  agents: AgentCostSummary[];
  models: Record<string, ModelBreakdown>;
  unpriced_models: string[];
  totals: {
    cost: number;
    requests: number;
    input_tokens: number;
    output_tokens: number;
  };
}

export interface BudgetPeriodStatus {
  limit: number | null;
  spent: number;
  remaining: number | null;
  percentUsed: number | null;
  resetsAt: string;
}

export interface BudgetStatus {
  agentId: string;
  daily: BudgetPeriodStatus;
  monthly: BudgetPeriodStatus;
  limitType: 'hard' | 'soft';
  blocked: boolean;
}
```

IMPORTANT: The proxy's cost-api.ts only maps periods 'hour', 'day'/'today', 'month', 'all'. It does NOT have 'week'. However, the CostAggregator DOES support 'week' (it filters to last 7 days). The cost-api handler has a switch/default that maps unknown periods to 'all'. For the "7 Days" switcher option, use period='all' and filter client-side by timestamp, OR accept that the 7-day view shows all-time data (acceptable for MVP since the proxy can be updated later). Check if adding 'week' to the cost-api switch statement is feasible — if the executor finds it's a simple one-line addition to `src/cost-api.ts`, add it. Otherwise, use 'all' and note the limitation.

Actually, re-reading cost-api.ts more carefully: the switch does NOT include 'week', but CostAggregator.getSummary() DOES handle 'week'. So add 'week' to the cost-api.ts switch statement (a 2-line change: `case 'week': period = periodParam; break;`). Add `src/cost-api.ts` to modified files.
