<objective>
Implement the approval queue backend so that when a policy flags a request for human approval, the proxy returns HTTP 202 with a polling URL instead of forwarding. Agents poll for decisions, and approved requests are re-sent with a single-use approval token.

Purpose: Enable human-in-the-loop governance for sensitive agent operations. This is the backend that Phase 14 (Approval Queue UI) will build on top of. The approval flow must work even before the dashboard exists — approvals can be resolved via direct DB updates or API calls.

Output: ApprovalManager, approval timeout checker, `require_approval` policy type, polling endpoint, token validation, and integration into the server request pipeline.
</objective>

    ```typescript
    import type postgres from 'postgres';

    export class ApprovalManager {
      constructor(private sql: postgres.Sql) {}

      // Create a new approval request in the database
      // Returns the approval request ID and polling URL
      async createApprovalRequest(params: {
        agentId: string;
        provider: string;
        model?: string;
        targetPath: string;
        policyName: string;
        policyRule?: string;
        estimatedCost?: number;
        requestSummary: string;      // Truncated preview (first ~500 chars of content)
        requestPayload?: unknown;     // Full payload (only if store_payload=true)
        timeoutSeconds: number;
      }): Promise<{ id: string; pollingUrl: string; expiresAt: string }> {
        const expiresAt = new Date(Date.now() + params.timeoutSeconds * 1000);
        const [row] = await this.sql`
          INSERT INTO approval_requests (
            agent_id, provider, model, target_path, policy_name, policy_rule,
            estimated_cost, request_summary, request_payload,
            timeout_seconds, expires_at
          ) VALUES (
            ${params.agentId}, ${params.provider}, ${params.model ?? null},
            ${params.targetPath}, ${params.policyName}, ${params.policyRule ?? null},
            ${params.estimatedCost ?? null}, ${params.requestSummary},
            ${params.requestPayload ? JSON.stringify(params.requestPayload) : null},
            ${params.timeoutSeconds}, ${expiresAt}
          ) RETURNING id, expires_at
        `;
        return {
          id: row.id,
          pollingUrl: `/api/approvals/${row.id}`,
          expiresAt: row.expires_at.toISOString(),
        };
      }

      // Get approval status for polling
      // Returns status and approval token (if approved)
      async getApprovalStatus(id: string): Promise<{
        id: string;
        status: 'pending' | 'approved' | 'denied' | 'denied_timeout';
        approvalToken?: string;
        decidedAt?: string;
        expiresAt: string;
      } | null> {
        const [row] = await this.sql`
          SELECT id, status, approval_token, decided_at, expires_at
          FROM approval_requests
          WHERE id = ${id}
        `;
        if (!row) return null;
        return {
          id: row.id,
          status: row.status,
          approvalToken: row.status === 'approved' ? row.approval_token : undefined,
          decidedAt: row.decided_at?.toISOString(),
          expiresAt: row.expires_at.toISOString(),
        };
      }

      // Validate an approval token for re-send flow
      // Returns the original request context if valid, null if invalid/expired/used
      async validateAndConsumeToken(token: string): Promise<{
        agentId: string;
        policyName: string;
        targetPath: string;
      } | null> {
        // Atomic: find the token and mark it as used in one query
        const [row] = await this.sql`
          UPDATE approval_requests
          SET token_used = true
          WHERE approval_token = ${token}
            AND status = 'approved'
            AND token_used = false
          RETURNING agent_id, policy_name, target_path
        `;
        if (!row) return null;
        return {
          agentId: row.agent_id,
          policyName: row.policy_name,
          targetPath: row.target_path,
        };
      }

      // Approve a request (used by dashboard in Phase 14, or direct DB/API)
      async approveRequest(id: string, decidedBy: string, notes?: string): Promise<boolean> {
        const approvalToken = crypto.randomUUID();
        const result = await this.sql`
          UPDATE approval_requests
          SET status = 'approved',
              decided_by = ${decidedBy},
              decision_notes = ${notes ?? null},
              decided_at = NOW(),
              approval_token = ${approvalToken}
          WHERE id = ${id} AND status = 'pending'
        `;
        return result.count > 0;
      }

      // Deny a request
      async denyRequest(id: string, decidedBy: string, notes?: string): Promise<boolean> {
        const result = await this.sql`
          UPDATE approval_requests
          SET status = 'denied',
              decided_by = ${decidedBy},
              decision_notes = ${notes ?? null},
              decided_at = NOW()
          WHERE id = ${id} AND status = 'pending'
        `;
        return result.count > 0;
      }
    }
    ```

    **Create `src/approval-timeout.ts`** — ApprovalTimeoutChecker:

    ```typescript
    export class ApprovalTimeoutChecker {
      private interval: ReturnType<typeof setInterval> | null = null;

      constructor(private sql: postgres.Sql) {}

      // Start checking for expired approvals every 30 seconds
      start(): void {
        this.interval = setInterval(() => this.expireTimedOut(), 30_000);
        this.interval.unref(); // Don't keep process alive
      }

      // Mark all expired pending approvals as denied_timeout
      async expireTimedOut(): Promise<number> {
        const result = await this.sql`
          UPDATE approval_requests
          SET status = 'denied_timeout', decided_at = NOW()
          WHERE status = 'pending' AND expires_at < NOW()
        `;
        if (result.count > 0) {
          console.log(`[govyn] Auto-denied ${result.count} expired approval request(s)`);
        }
        return result.count;
      }

      stop(): void {
        if (this.interval) {
          clearInterval(this.interval);
          this.interval = null;
        }
      }
    }
    ```
