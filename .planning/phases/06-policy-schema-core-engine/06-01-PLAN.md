```typescript
/** The six policy types supported by Govyn (Phase 6 defines skeletons; Phase 7 adds evaluators) */
export type PolicyType = 'block' | 'rate_limit' | 'budget_limit' | 'content_filter' | 'time_window' | 'model_route';

/** Policy scope — determines which requests a policy applies to */
export interface PolicyScope {
  /** 'global' applies to all requests; 'agent:{name}' scopes to a specific agent; 'target:{provider}' scopes to a provider */
  level: 'global' | 'agent' | 'target';
  /** The agent name (when level='agent') or target provider (when level='target'), undefined for global */
  value?: string;
}

/** Base fields shared by all policy definitions */
export interface PolicyBase {
  /** Unique policy name (must be unique across all loaded policies) */
  name: string;
  /** Human-readable description of what this policy does */
  description?: string;
  /** Whether this policy is active (default: true) */
  enabled: boolean;
  /** The type of policy rule */
  type: PolicyType;
  /** Scoping: who/what this policy applies to */
  scope: PolicyScope;
}

/** Block policy — skeleton, full match criteria added in Phase 7 */
export interface BlockPolicy extends PolicyBase {
  type: 'block';
  match?: Record<string, unknown>;
  message?: string;
}

/** Rate limit policy — skeleton */
export interface RateLimitPolicy extends PolicyBase {
  type: 'rate_limit';
  limit?: number;
  window_seconds?: number;
}

/** Budget limit policy — skeleton */
export interface BudgetLimitPolicy extends PolicyBase {
  type: 'budget_limit';
  limit?: number;
  period?: string;
}

/** Content filter policy — skeleton */
export interface ContentFilterPolicy extends PolicyBase {
  type: 'content_filter';
  patterns?: string[];
}

/** Time window policy — skeleton */
export interface TimeWindowPolicy extends PolicyBase {
  type: 'time_window';
  allow_hours?: { start: string; end: string };
  days?: string[];
}

/** Model route policy — skeleton per ADR-021 */
export interface ModelRoutePolicy extends PolicyBase {
  type: 'model_route';
  rules?: unknown[];
  model_aliases?: Record<string, string>;
}

/** Union type of all policy variants */
export type Policy = BlockPolicy | RateLimitPolicy | BudgetLimitPolicy | ContentFilterPolicy | TimeWindowPolicy | ModelRoutePolicy;

/** Top-level policy file structure */
export interface PolicyFile {
  /** Schema version — must be 1 per ADR-018 */
  version: number;
  /** Array of policy definitions */
  policies: Policy[];
}

/** A policy parse error with location information */
export interface PolicyParseError {
  /** Human-readable error message */
  message: string;
  /** Line number in the YAML file (1-indexed), if available */
  line?: number;
  /** Column number (1-indexed), if available */
  column?: number;
  /** The policy name this error relates to, if applicable */
  policyName?: string;
}

/** Result of parsing a policy file */
export interface PolicyParseResult {
  /** Whether parsing succeeded without errors */
  success: boolean;
  /** Parsed policies (empty if success=false) */
  policies: Policy[];
  /** Validation errors (empty if success=true) */
  errors: PolicyParseError[];
  /** Non-fatal warnings */
  warnings: PolicyParseError[];
}
```

**GREEN phase of TDD: Write the parser implementation to make all tests pass.**

Create `src/policy-parser.ts` with these exports:

**`parsePolicies(yamlString: string): PolicyParseResult`**

Implementation approach:
1. Use the `yaml` library's `parseDocument()` (NOT `parse()`) to get a Document with source map information (CST nodes have `range` properties for line numbers). The `yaml` package is already a dependency (used in src/config.ts).
2. Validate top-level structure:
   - Check `version` field exists and equals `1` (per ADR-018). If missing or wrong, return error with line number.
   - Check `policies` is an array. If missing, return error.
3. For each policy entry, validate:
   - `name` is a non-empty string (extract line number from Document node for error reporting)
   - `type` is one of the six valid PolicyType values
   - `scope` is either missing (defaults to `{level:'global'}`) or a valid scope string:
     - `"global"` → `{level:'global'}`
     - `"agent:<name>"` → `{level:'agent', value:'<name>'}`
     - `"target:<provider>"` → `{level:'target', value:'<provider>'}`
   - `enabled` defaults to `true` if not specified
   - No duplicate `name` values across all policies
4. Return `PolicyParseResult` with:
   - `success: true` only if zero errors
   - `policies` array with properly typed Policy objects
   - `errors` array with line numbers extracted from Document source map
   - `warnings` for non-fatal issues (e.g., empty policies array)

**`parsePoliciesFromFile(filePath: string): PolicyParseResult`**
- Read file with `fs.readFileSync()`, call `parsePolicies()` on the contents
- If file read fails, return error result with file read error message

**Line number extraction approach:**
Use `yaml` library's `parseDocument()` which returns a `Document`. Use `doc.contents` to walk YAML nodes. Each YAML node has a `range` property `[start, valueEnd, nodeEnd]` — convert the character offset to a line number by counting newlines in the source string up to `range[0]`. Create a helper `offsetToLine(source: string, offset: number): number` for this.

**Important patterns to follow:**
- Match existing code style in src/config.ts (same yaml import pattern)
- Return errors, don't throw — the parser should be safe to call and always return a result
- Policy type-specific fields (match, limit, patterns, etc.) are stored as-is in Phase 6 — validation of type-specific fields happens in Phase 7 when evaluators are added
