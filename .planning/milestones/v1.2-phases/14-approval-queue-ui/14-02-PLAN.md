  ```typescript
  async function handleConfirm(id: string, notes: string) {
    const endpoint = modalState.action === 'approve' ? 'approve' : 'deny';
    const response = await apiFetch(`/api/approvals/${id}/${endpoint}`, {
      method: 'POST',
      body: JSON.stringify({ decided_by: 'dashboard', notes: notes || undefined }),
    });
    if (!response.ok) {
      const err = await response.json().catch(() => ({ error: 'Unknown error' }));
      toast.error(`Failed to ${endpoint}: ${err.error || response.statusText}`);
      throw new Error('Action failed'); // keeps modal open
    }
    toast.success(
      modalState.action === 'approve'
        ? 'Request approved successfully'
        : 'Request denied'
    );
    setModalState({ open: false, approval: null, action: 'approve' });
    refetch(); // refresh the approval list
  }
  ```
