**Create `.github/workflows/ci.yml`:**
```yaml
name: CI
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  lint-and-typecheck:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - run: npm run lint
      - run: npm run typecheck

  test:
    runs-on: ubuntu-latest
    needs: lint-and-typecheck
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - run: npm test -- --reporter=verbose
      - name: Upload test results
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: test-results
          path: test-results/

  build:
    runs-on: ubuntu-latest
    needs: test
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - run: npm run build
      - name: Verify dist output
        run: ls -la dist/ && test -f dist/cli.js && test -f dist/index.js

  docker:
    runs-on: ubuntu-latest
    needs: build
    steps:
      - uses: actions/checkout@v4
      - name: Build Docker image
        run: docker build -t govyn:ci .
      - name: Check image size
        run: |
          SIZE=$(docker image inspect govyn:ci --format='{{.Size}}')
          MB=$((SIZE / 1048576))
          echo "Image size: ${MB}MB"
          if [ "$MB" -gt 100 ]; then
            echo "FAIL: Image exceeds 100MB"
            exit 1
          fi
      - name: Smoke test
        run: |
          docker run --rm govyn:ci --help
```

The pipeline has 4 stages: lint+typecheck -> test -> build -> docker. Each stage depends on the previous. This ensures fast feedback (lint fails first) and complete validation.

**Add conditional publish jobs** that trigger on version tags (`v*`). These structurally satisfy PACK-05's "publish npm + Docker image" requirement while remaining inert until credentials are configured:

Update the `on:` trigger in ci.yml to also include:
```yaml
on:
  push:
    branches: [main]
    tags: ['v*']
  pull_request:
    branches: [main]
```

Add two publish jobs after the docker job:

```yaml
  publish-npm:
    if: startsWith(github.ref, 'refs/tags/v')
    runs-on: ubuntu-latest
    needs: [test, build]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
          registry-url: https://registry.npmjs.org
      - run: npm ci
      - run: npm run build
      - run: npm publish --access public
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}

  publish-docker:
    if: startsWith(github.ref, 'refs/tags/v')
    runs-on: ubuntu-latest
    needs: [test, docker]
    steps:
      - uses: actions/checkout@v4
      - name: Log in to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}
      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          push: true
          tags: govyn/govyn:${{ github.ref_name }},govyn/govyn:latest
```

These jobs only run when a `v*` tag is pushed and require `NPM_TOKEN`, `DOCKERHUB_USERNAME`, and `DOCKERHUB_TOKEN` secrets to be configured. Until those secrets exist, the jobs will fail gracefully (they only trigger on tags, not on regular pushes/PRs). Phase 17 will configure the actual secrets and first release.
